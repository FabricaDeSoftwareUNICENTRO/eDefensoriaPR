CREATE TABLE ASSISTIDO (
  cpf INTEGER UNSIGNED NOT NULL,
  BENS_idBens INTEGER UNSIGNED NOT NULL,
  ESTADO_CIVIL_idESTADO_CIVIL INTEGER UNSIGNED NOT NULL,
  PARENTE_cpf INTEGER UNSIGNED NOT NULL,
  CONTATO_cpf INTEGER UNSIGNED NOT NULL,
  PROCESSO_idPROCESSO INTEGER UNSIGNED NOT NULL,
  nome VARCHAR NULL,
  telefone INTEGER UNSIGNED NULL,
  dataNascimento DATE NULL,
  naturalidade TEXT NULL,
  profissao TEXT NULL,
  localTrabalho TEXT NULL,
  idade INTEGER UNSIGNED NULL,
  genero TEXT NULL,
  rg INTEGER UNSIGNED NULL,
  PRIMARY KEY(cpf),
  INDEX ASSISTIDO_FKIndex1(PARENTE_cpf),
  INDEX ASSISTIDO_FKIndex3(PROCESSO_idPROCESSO),
  INDEX ASSISTIDO_FKIndex4(CONTATO_cpf),
  INDEX ASSISTIDO_FKIndex5(BENS_idBens),
  INDEX ASSISTIDO_FKIndex6(ESTADO_CIVIL_idESTADO_CIVIL)
);

CREATE TABLE ATENDIMENTO (
  idATENDIMENTO INTEGER UNSIGNED NOT NULL AUTO_INCREMENT,
  PROCESSO_idPROCESSO INTEGER UNSIGNED NOT NULL,
  FUNCIONARIO_idFUNCIONARIO INTEGER UNSIGNED NOT NULL,
  PRIMARY KEY(idATENDIMENTO),
  INDEX ATENDIMENTO_FKIndex1(FUNCIONARIO_idFUNCIONARIO),
  INDEX ATENDIMENTO_FKIndex2(PROCESSO_idPROCESSO)
);

CREATE TABLE BENS (
  idBens INTEGER UNSIGNED NOT NULL AUTO_INCREMENT,
  descricao TEXT NULL,
  PRIMARY KEY(idBens)
);

CREATE TABLE CONTATO (
  cpf INTEGER UNSIGNED NOT NULL AUTO_INCREMENT,
  nome VARCHAR NULL,
  endereco VARCHAR NULL,
  PRIMARY KEY(cpf)
);

CREATE TABLE ESTADO_CIVIL (
  idESTADO_CIVIL INTEGER UNSIGNED NOT NULL AUTO_INCREMENT,
  descricao INTEGER UNSIGNED NULL,
  PRIMARY KEY(idESTADO_CIVIL)
);

CREATE TABLE FUNCIONARIO (
  idFUNCIONARIO INTEGER UNSIGNED NOT NULL AUTO_INCREMENT,
  nome INTEGER UNSIGNED NULL,
  funcao TEXT NULL,
  PRIMARY KEY(idFUNCIONARIO)
);

CREATE TABLE HABITACAO (
  cep INTEGER UNSIGNED NOT NULL,
  SITUACAO_HABITACAO_idSituacaoHabitacao INTEGER UNSIGNED NOT NULL,
  enderecoDomiciliar TEXT NOT NULL,
  cidadeDomiciliar TEXT NULL,
  bairro TEXT NULL,
  INDEX HABITACAO_FKIndex1(SITUACAO_HABITACAO_idSituacaoHabitacao)
);

CREATE TABLE LOCAL_TRABALHO (
  cep INTEGER UNSIGNED NOT NULL,
  endereco TEXT NULL,
  cidade TEXT NULL,
  bairro TEXT NULL
);

CREATE TABLE PARENTE (
  cpf INTEGER UNSIGNED NOT NULL,
  endereco VARCHAR NULL,
  telefone INTEGER UNSIGNED NULL,
  nome VARCHAR NULL,
  PRIMARY KEY(cpf)
);

CREATE TABLE PARTE_CONTRARIA (
  cpf INTEGER UNSIGNED NOT NULL ,
  PROCESSO_idPROCESSO INTEGER UNSIGNED NOT NULL,
  nome VARCHAR(255) NULL,
  endereco VARCHAR(255) NULL,
  cidade TEXT NULL,
  remuneracao INTEGER UNSIGNED NULL,
  cep INTEGER UNSIGNED NULL,
  rg TEXT NULL,
  profissao TEXT NULL,
  bairro TEXT NULL,
  telefone INT NULL,
  PRIMARY KEY(cpf),
  INDEX PARTE_CONTRARIA_FKIndex1(PROCESSO_idPROCESSO)
);

CREATE TABLE PENDENCIA (
  descricao TEXT NOT NULL,
  PROCESSO_idPROCESSO INTEGER UNSIGNED NOT NULL,
  INDEX PENDENCIA_FKIndex1(PROCESSO_idPROCESSO)
);

CREATE TABLE PROCESSO (
  idPROCESSO INTEGER UNSIGNED NOT NULL AUTO_INCREMENT,
  SITUACAO_PROCESSO_idSituacaoProcesso INTEGER UNSIGNED NOT NULL,
  descricao TEXT NULL,
  PRIMARY KEY(idPROCESSO),
  INDEX PROCESSO_FKIndex2(SITUACAO_PROCESSO_idSituacaoProcesso)
);

CREATE TABLE SITUACAO_HABITACAO (
  idSituacaoHabitacao INTEGER UNSIGNED NOT NULL AUTO_INCREMENT,
  descricao TEXT NULL,
  PRIMARY KEY(idSituacaoHabitacao)
);

CREATE TABLE SITUACAO_PROCESSO (
  idSituacaoProcesso INTEGER UNSIGNED NOT NULL AUTO_INCREMENT,
  descricao TEXT NULL,
  PRIMARY KEY(idSituacaoProcesso)
);
